\begin{frame}{Models}
\vspace{0.1cm}
\begin{block}{BiLSTM}
Bidirectional LSTM trained from scratch. Captures sequential context in both directions.
\end{block}
\vspace{0.15cm}
\begin{block}{BERT \normalfont\footnotesize\texttt{bert-base-uncased}}
Transformer encoder pretrained via masked language modeling. Fine-tuned end-to-end.
\end{block}
\vspace{0.15cm}
\begin{block}{RoBERTa \normalfont\footnotesize\texttt{roberta-base}}
Optimized BERT: removes next sentence prediction, uses dynamic masking and more data.
\end{block}
\end{frame}

\begin{frame}{Atakan Gül — Media Bias Detection}
\vspace{0.2cm}
\begin{columns}[t]
\column{0.48\textwidth}
    {\small\textbf{Task}}\par
    \vspace{0.1cm}
    {\small Classify a news sentence as \textit{biased} or \textit{not biased}.}\par
\column{0.52\textwidth}
    {\small\textbf{Dataset: BABE}}\par
    \vspace{0.1cm}
    \begin{itemize}\small
        \item 4,121 expert-annotated sentences
        \item Diverse news outlets
        \item 3,121 train \textbf{/} 1,000 test
    \end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Atakan — Results}
\vspace{0.1cm}
\begin{columns}[c]
\column{0.44\textwidth}
    \begin{tabular}{lcc}
    \toprule
    \textbf{Model} & \textbf{Acc.} & \textbf{F1} \\
    \midrule
    BiLSTM   & 0.656 & 0.651 \\
    BERT     & 0.826 & 0.827 \\
    RoBERTa  & \textbf{0.846} & \textbf{0.846} \\
    \bottomrule
    \end{tabular}
    \vspace{0.35cm}
    \begin{itemize}\small
        \item RoBERTa best: +19\% over BiLSTM
        \item BERT close behind
        \item BiLSTM struggles on subtle bias
    \end{itemize}
\column{0.56\textwidth}
    \includegraphics[width=\linewidth]{model_comparison.png}
\end{columns}
\end{frame}

\begin{frame}{Atakan — Conclusion}
\vspace{0.5cm}
\begin{itemize}\setlength\itemsep{0.4cm}
    \item Media bias is expressed through framing and tone — not facts
    \item Pretraining is critical: transformers capture patterns BiLSTM cannot
    \item RoBERTa's optimized pretraining gives it a clear edge over BERT
\end{itemize}
\end{frame}
