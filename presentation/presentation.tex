%% ── Slide 3: Models
\begin{frame}{Models}
\begin{block}{BiLSTM}
Bidirectional LSTM trained from scratch. Captures sequential context in both directions.
\end{block}
\begin{block}{BERT \quad \small\normalfont\texttt{bert-base-uncased}}
Transformer encoder pretrained via masked language modeling. Fine-tuned end-to-end.
\end{block}
\begin{block}{RoBERTa \quad \small\normalfont\texttt{roberta-base}}
Optimized BERT: removes next sentence prediction, uses dynamic masking and more data.
\end{block}
\end{frame}

\begin{frame}{Atakan Gül — Media Bias Detection}
\begin{columns}[t]
\column{0.5\textwidth}
\textbf{Task}\\[0.2cm]
Classify a news sentence as \textit{biased} or \textit{not biased}.
\column{0.5\textwidth}
\textbf{Dataset: BABE}\\[0.2cm]
\begin{itemize}
    \item 4,121 expert-annotated sentences
    \item Diverse news outlets
    \item 3,121 train / 1,000 test
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Atakan — Results}
\begin{columns}[c]
\column{0.45\textwidth}
\begin{tabular}{lcc}
\toprule
\textbf{Model} & \textbf{Acc.} & \textbf{F1} \\
\midrule
BiLSTM   & 0.656 & 0.651 \\
BERT     & 0.826 & 0.827 \\
RoBERTa  & \textbf{0.846} & \textbf{0.846} \\
\bottomrule
\end{tabular}
\vspace{0.3cm}
\begin{itemize}
    \item RoBERTa best: +19\% over BiLSTM
    \item BERT close behind
    \item BiLSTM struggles on subtle bias
\end{itemize}
\column{0.55\textwidth}
\includegraphics[width=\linewidth]{model_comparison.png}
\end{columns}
\end{frame}

\begin{frame}{Atakan — Prediction Examples}
\small
\begin{tabular}{p{6.8cm}cccc}
\toprule
\textbf{Sentence} & \textbf{True} & \textbf{LSTM} & \textbf{BERT} & \textbf{RoBERTa} \\
\midrule
\textit{``Nearly 78\% of Americans report concern about the climate...''} & NB & NB\,\checkmark & NB\,\checkmark & NB\,\checkmark \\[0.25cm]
\textit{``Companies like Nike and Google made Juneteenth a paid holiday.''} & NB & B\,\(\times\) & NB\,\checkmark & NB\,\checkmark \\[0.25cm]
\textit{``None of Biden's Democrat counterparts challenged his claim of economic selflessness.''} & B & B\,\checkmark & NB\,\(\times\) & B\,\checkmark \\
\bottomrule
\end{tabular}
\vspace{0.15cm}
{\footnotesize NB = Not Biased \quad B = Biased\\[0.1cm] \checkmark\,= correct \quad $\times$\,= incorrect}
\end{frame}

\begin{frame}{Atakan — Conclusion}
\begin{itemize}\setlength\itemsep{0.5cm}
    \item Media bias is expressed through framing and tone — not facts
    \item Pretraining is critical: transformers capture patterns BiLSTM cannot
    \item RoBERTa's optimized pretraining gives it a clear edge over BERT
\end{itemize}
\end{frame}
