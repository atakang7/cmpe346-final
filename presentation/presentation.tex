%% ── Slide 3: Atakan — Task & Dataset ────────────────────────
\begin{frame}{Media Bias Detection in News}
\textbf{Task:} Given a news sentence, classify it as \textit{biased} or \textit{not biased}.

\vspace{0.4cm}
\textbf{Dataset: BABE} (Bias Annotations By Experts)
\begin{itemize}
    \item 4,121 news sentences from diverse outlets
    \item Expert-annotated binary labels
    \item 3,121 train / 1,000 test
\end{itemize}

\vspace{0.4cm}
\textit{Bias is expressed through framing and tone — not false facts. This makes it a genuinely hard NLP task.}
\end{frame}

%% ── Slide 4: Atakan — Methodology ──────────────────────────
\begin{frame}{Models}
\begin{block}{BiLSTM}
Bidirectional recurrent network. Trained from scratch. Captures sequential patterns in both directions.
\end{block}
\begin{block}{BERT}
Transformer encoder pretrained on masked language modeling. Fine-tuned on BABE. Uses \texttt{[CLS]} token for classification.
\end{block}
\begin{block}{RoBERTa}
Optimized BERT variant. Removes next sentence prediction, uses dynamic masking and more data. Consistently stronger on downstream tasks.
\end{block}
\end{frame}
