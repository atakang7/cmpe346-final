%% ══ ATAKAN ══════════════════════════════════

\begin{frame}{Atakan Gül — Media Bias Detection}
\begin{columns}[t]
\column{0.5\textwidth}
\textbf{Task}\\[0.2cm]
Classify a news sentence as \textit{biased} or \textit{not biased}.
\column{0.5\textwidth}
\textbf{Dataset: BABE}\\[0.2cm]
\begin{itemize}
    \item 4,121 expert-annotated sentences
    \item Diverse news outlets
    \item 3,121 train / 1,000 test
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Atakan — Results}
\begin{columns}[c]
\column{0.45\textwidth}
\begin{tabular}{lcc}
\toprule
\textbf{Model} & \textbf{Acc.} & \textbf{W-F1} \\
\midrule
BiLSTM   & 0.6560 & 0.6511 \\
BERT     & 0.8260 & 0.8266 \\
RoBERTa  & \textbf{0.8460} & \textbf{0.8464} \\
\bottomrule
\end{tabular}
\vspace{0.3cm}
\begin{itemize}
    \item RoBERTa best: +19pp accuracy over BiLSTM
    \item BERT close behind
    \item BiLSTM struggles on subtle bias
\end{itemize}
\column{0.55\textwidth}
\includegraphics[width=\linewidth]{model_comparison.png}
\end{columns}
\end{frame}

\begin{frame}{Atakan — Prediction Examples}
\resizebox{\textwidth}{!}{%
\begin{tabular}{p{8cm}cccc}
\toprule
\textbf{Sentence} & \textbf{True} & \textbf{LSTM} & \textbf{BERT} & \textbf{RoBERTa} \\
\midrule
\textit{``Companies like Nike and Google made Juneteenth a paid holiday.''} & NB & B\,$\times$ & NB\,\checkmark & NB\,\checkmark \\[0.4cm]
\textit{``None of Biden's Democrat counterparts challenged his claim of economic selflessness.''} & B & B\,\checkmark & NB\,$\times$ & B\,\checkmark \\
\bottomrule
\end{tabular}%
}
\vspace{0.2cm}
{\footnotesize NB = Not Biased \quad B = Biased \quad \checkmark\,= correct \quad $\times$\,= incorrect}
\end{frame}

\begin{frame}{Atakan — Conclusion}
\begin{itemize}\setlength\itemsep{0.5cm}
    \item Pretraining is critical: transformers capture subtle bias patterns BiLSTM cannot
    \item RoBERTa is the best model — 84.60\% accuracy, 0.8464 weighted F1
\end{itemize}
\end{frame}

%% ══ STUDENT 3 ════════════════════════════════

\begin{frame}{Talha Demir — Ghost-Job Risk Classification}
\begin{columns}[t]
\column{0.5\textwidth}
\textbf{Task}\\[0.2cm]
Classify job postings by visible ghost-job risk evidence.
\column{0.5\textwidth}
\textbf{Dataset: Zenodo 20321172}\\[0.2cm]
\begin{itemize}
    \item unified\_core job postings
    \item Evidence-based target correction
    \item 3:1 realistic-to-ghost downsampled splits
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Talha — Results}
\begin{columns}[c]
\column{0.45\textwidth}
\begin{tabular}{lcc}
\toprule
\textbf{Model} & \textbf{Acc.} & \textbf{W-F1} \\
\midrule
BiLSTM   & 0.8636 & 0.8569 \\
BERT     & 0.9015 & 0.9029 \\
RoBERTa  & \textbf{0.9621} & \textbf{0.9610} \\
\bottomrule
\end{tabular}
\vspace{0.3cm}
\begin{itemize}
    \item RoBERTa best overall
    \item BERT improves likely-ghost recall over BiLSTM
\end{itemize}
\column{0.55\textwidth}
\includegraphics[width=\linewidth]{talha_model_comparison.png}
\end{columns}
\end{frame}

\begin{frame}{Talha — Prediction Examples}
\resizebox{\textwidth}{!}{%
\begin{tabular}{p{8cm}ccc}
\toprule
\textbf{Posting signal} & \textbf{True} & \textbf{RoBERTa} & \textbf{Why it matters} \\
\midrule
\textit{``This position is contingent upon contract award.''} & Ghost & Ghost\,\checkmark & Explicit contingency \\[0.35cm]
\textit{``Join our talent community for future opportunities.''} & Ghost & Ghost\,\checkmark & Pipeline-only wording \\[0.35cm]
\textit{``We are shaping the future of AI with an active engineering team.''} & Realistic & Ghost\,$\times$ & Hard phrase: harmless ``future'' context \\
\bottomrule
\end{tabular}%
}
\vspace{0.2cm}
{\footnotesize Ghost = likely ghost-risk \quad \checkmark\,= correct \quad $\times$\,= incorrect}
\end{frame}

\begin{frame}{Talha — Conclusion}
\begin{itemize}\setlength\itemsep{0.5cm}
    \item Dataset correction was essential because raw ghost labels were noisy
    \item Labels reflect visible text signals, not verified hiring outcomes
    \item RoBERTa reaches 96.21\% accuracy and 0.9610 weighted F1
\end{itemize}
\end{frame}
